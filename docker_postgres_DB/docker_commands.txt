# Docker start the container (must be inside the folder where the docker-compose file is)
docker-compose up -d

# Access the DB inside the docker container
docker exec -it ishango_postgres psql -U postgres

# List all DBs
\l

# Connect to a specific DB
\c (or \connect) ishango


## Once inside a specific DB
# List all tables
\dt

# Output the first few lines of a Table
TABLE ghana2022 LIMIT 5;


## To kill a process
# List all docker containers
docker ps (or docker container ls)

# Kill (you can also stop, it tries to kill with a grace period instead of killing immediatly)
docker kill ishango_postgres
